psycopg2 — conexão, query e segurança

psycopg2 é o driver mais popular para PostgreSQL em Python. Importante: nunca interpolar strings para montar SQL — use sempre parâmetros (%s) para evitar SQL Injection.

Python — psycopg2 completo e seguro
import psycopg2
from psycopg2.extras import RealDictCursor, execute_values
import os
from dotenv import load_dotenv

load_dotenv()  # carrega variáveis do arquivo .env

# ── Conectar ──────────────────────────────────────────────────────
conn = psycopg2.connect(os.getenv("DATABASE_URL"))
# ou com parâmetros individuais:
conn = psycopg2.connect(
    host=os.getenv("DB_HOST", "localhost"),
    port=os.getenv("DB_PORT", "5432"),
    dbname=os.getenv("DB_NAME"),
    user=os.getenv("DB_USER"),
    password=os.getenv("DB_PASSWORD"),
    sslmode="require",  # exige SSL em produção
)

# ── SELECT com parâmetros seguros ─────────────────────────────────
cur = conn.cursor()
cur.execute(
    "SELECT id, nome, saldo FROM contas WHERE saldo > %s",
    (1000,)   # sempre usar tupla, nunca f-string!
)
linhas = cur.fetchall()
print(linhas)

# ── RealDictCursor — retorna dicionários em vez de tuplas ─────────
cur = conn.cursor(cursor_factory=RealDictCursor)
cur.execute("SELECT * FROM pedidos WHERE status = %s LIMIT 5", ("entregue",))
for row in cur.fetchall():
    print(row["id"], row["total"])

# ── INSERT com valores múltiplos (execute_values é muito mais rápido)
dados = [(i, f"cliente_{i}", i * 100.0) for i in range(1, 1001)]
execute_values(
    cur,
    "INSERT INTO clientes_teste (id, nome, saldo) VALUES %s ON CONFLICT DO NOTHING",
    dados,
    page_size=200   # agrupa em batches de 200 para eficiência
)
conn.commit()

# ── Fechar recursos corretamente ──────────────────────────────────
cur.close()
conn.close()

# ── Forma mais segura: context manager ───────────────────────────
with psycopg2.connect(os.getenv("DATABASE_URL")) as conn:
    with conn.cursor() as cur:
        cur.execute("SELECT COUNT(*) FROM pedidos")
        total = cur.fetchone()[0]
        print(f"Total de pedidos: {total}")
    conn.commit()  # context manager fecha cursor, mas commit é manual
SQL Injection — nunca faça isso:
cur.execute(f"SELECT * FROM users WHERE nome = '{nome_usuario}'")

Se nome_usuario = "'; DROP TABLE users; --", a tabela será apagada. Sempre use parâmetros: cur.execute("SELECT * FROM users WHERE nome = %s", (nome_usuario,))

Cursor de servidor — para grandes volumes

O cursor padrão do psycopg2 carrega todos os resultados em memória. Para consultas que retornam milhões de linhas, use cursor de servidor (named cursor) — o banco envia os dados em lotes.

Python — cursor de servidor para tabelas grandes
with psycopg2.connect(os.getenv("DATABASE_URL")) as conn:
    # Named cursor = cursor de servidor (iterador, não carrega tudo)
    with conn.cursor("cursor_grande") as cur:
        cur.itersize = 10_000  # busca 10k linhas por vez do servidor
        cur.execute("SELECT id, total, status FROM pedidos ORDER BY id")
        processados = 0
        for row in cur:   # itera sem carregar tudo na memória
            processados += 1
            if processados % 50_000 == 0:
                print(f"Processados: {processados:,}")
        print(f"Total: {processados:,} linhas")

SQLAlchemy — engine, connection pool e Core

SQLAlchemy é a biblioteca de banco de dados mais completa do ecossistema Python. O engine gerencia um pool de conexões automaticamente — você não precisa abrir e fechar conexões manualmente a cada operação.

Python — SQLAlchemy Core: engine e queries
from sqlalchemy import create_engine, text
import os

# Engine com connection pool automático
engine = create_engine(
    os.getenv("DATABASE_URL"),
    pool_size=5,        # conexões permanentes no pool
    max_overflow=10,    # conexões extras em pico
    pool_timeout=30,    # segundos para aguardar conexão disponível
    pool_recycle=1800,  # reconecta após 30min (evita conexões stale)
    echo=False,         # True = loga todo SQL (dev only)
)

# ── SELECT com parâmetros nomeados ────────────────────────────────
with engine.connect() as conn:
    resultado = conn.execute(
        text("SELECT id, nome, saldo FROM contas WHERE saldo > :minimo"),
        {"minimo": 1000}
    )
    for row in resultado:
        print(row.id, row.nome, row.saldo)

# ── INSERT com transação explícita ────────────────────────────────
with engine.begin() as conn:  # begin() faz COMMIT automático ao sair
    conn.execute(
        text("INSERT INTO log_eventos (evento) VALUES (:evento)"),
        [{"evento": "pipeline iniciado"}, {"evento": "pipeline concluído"}]
    )
    # COMMIT automático ao sair do with — ROLLBACK se lançar exceção

# ── SQLAlchemy ORM — definir modelo e consultar ───────────────────
from sqlalchemy.orm import DeclarativeBase, Mapped, mapped_column, Session
from sqlalchemy import Integer, String, Numeric

class Base(DeclarativeBase): pass

class Conta(Base):
    __tablename__ = "contas"
    id:    Mapped[int]   = mapped_column(Integer, primary_key=True)
    nome:  Mapped[str]   = mapped_column(String(100))
    saldo: Mapped[float] = mapped_column(Numeric(12, 2))

# Consultar com ORM
with Session(engine) as session:
    contas_ricas = session.query(Conta).filter(Conta.saldo > 2000).all()
    for c in contas_ricas:
        print(f"{c.nome}: R${c.saldo}")
pandas + SQLAlchemy — read_sql e to_sql

A combinação pandas + SQLAlchemy é a mais produtiva para pipelines de análise e feature engineering. read_sql retorna um DataFrame diretamente da query; to_sql grava o DataFrame de volta no banco.

Python — pandas read_sql e to_sql completo
import pandas as pd
from sqlalchemy import create_engine, text
import os

engine = create_engine(os.getenv("DATABASE_URL"))

# ── read_sql: query → DataFrame ───────────────────────────────────
df_pedidos = pd.read_sql(
    sql="""
        SELECT p.id, p.cliente_id, p.total, p.status, p.criado_em,
               c.nome AS cliente_nome
        FROM pedidos p
        JOIN clientes c ON c.id = p.cliente_id
        WHERE p.status != 'cancelado'
        ORDER BY p.criado_em DESC
    """,
    con=engine,
    parse_dates=["criado_em"],  # converte coluna para datetime automaticamente
)
print(f"Linhas: {len(df_pedidos)} | Tipos:\n{df_pedidos.dtypes}")

# ── read_sql com parâmetro seguro ─────────────────────────────────
status_filtro = "entregue"
df_entregues = pd.read_sql(
    sql=text("SELECT * FROM pedidos WHERE status = :status"),
    con=engine,
    params={"status": status_filtro},
)

# ── Transformação pandas ──────────────────────────────────────────
df_features = df_pedidos.groupby("cliente_id").agg(
    total_pedidos  =("id",        "count"),
    ticket_medio   =("total",     "mean"),
    total_gasto    =("total",     "sum"),
    ultimo_pedido  =("criado_em", "max"),
    cliente_nome   =("cliente_nome", "first"),
).reset_index()

df_features["dias_inativo"] = (
    pd.Timestamp("today") - df_features["ultimo_pedido"]
).dt.days
df_features["ticket_medio"] = df_features["ticket_medio"].round(2)
df_features["gerado_em"]    = pd.Timestamp("now")

print(df_features.head())

# ── to_sql: DataFrame → banco ─────────────────────────────────────
df_features.to_sql(
    name="features_cliente",
    con=engine,
    if_exists="replace",   # "replace" | "append" | "fail"
    index=False,
    dtype={                   # forçar tipos específicos
        "cliente_id":   "INTEGER",
        "ticket_medio": "NUMERIC(10,2)",
        "total_gasto":  "NUMERIC(12,2)",
        "dias_inativo": "INTEGER",
    },
    chunksize=1000,          # insere em lotes de 1000 linhas
    method="multi",          # INSERT múltiplo por batch (mais rápido)
)
print(f"Features de {len(df_features)} clientes gravadas no banco ✓")
Parâmetro if_existsComportamentoQuando usar
"fail"Lança erro se a tabela já existeCriação inicial — previne sobrescrever dados
"replace"Apaga e recria a tabela antes de inserirTabelas de features reprocessadas diariamente
"append"Insere dados sem alterar a tabelaLogs, eventos, dados incrementais
to_sql com method="multi" e chunksize=1000 é significativamente mais rápido que o padrão (um INSERT por linha). Para volumes muito grandes (>100k linhas), considere COPY via psycopg2 ou a extensão pgcopy.

Boas práticas de segurança e conexão
✓ Faça isso
  • Armazene credenciais em variáveis de ambiente ou secrets manager
  • Use sempre parâmetros (%s, :nome) — nunca interpole strings
  • Habilite sslmode="require" em produção
  • Feche cursores e conexões com context manager (with)
  • Use connection pool — nunca abra uma conexão por query
  • Adicione .env ao .gitignore
  • Limite permissões do usuário de banco (princípio do menor privilégio)
✗ Nunca faça isso
  • Colocar senha hardcoded no código Python
  • Usar f-string para montar SQL com dados do usuário
  • Commitar arquivo .env no Git
  • Usar usuário postgres (superusuário) em produção
  • Abrir e fechar conexões dentro de loops (muito lento)
  • Ignorar erros de SSL em produção
  • Deixar conexões abertas indefinidamente sem pool_recycle
Python — padrão seguro de conexão com .env
# db.py — módulo centralizado de conexão
from sqlalchemy import create_engine
from dotenv import load_dotenv
import os

load_dotenv()

def get_engine(pool_size=5):
    """Cria e retorna um engine SQLAlchemy configurado de forma segura."""
    url = os.getenv("DATABASE_URL")
    if not url:
        raise EnvironmentError("DATABASE_URL não definida. Verifique o arquivo .env")
    return create_engine(
        url,
        pool_size=pool_size,
        max_overflow=10,
        pool_recycle=1800,
        connect_args={"sslmode": os.getenv("DB_SSLMODE", "prefer")},
    )

Pipeline completo — BD → pandas → transformação → carga
Python — pipeline_bd_pandas.py
"""
Pipeline: PostgreSQL → pandas → feature engineering → carga de volta no BD
"""
import pandas as pd
from sqlalchemy import create_engine, text
from dotenv import load_dotenv
import os, json
from datetime import datetime

load_dotenv()
engine = create_engine(os.getenv("DATABASE_URL"))
log = []

def etapa(nome, fn):
    """Executa uma etapa, mede tempo e registra no log."""
    import time
    t0 = time.perf_counter()
    resultado = fn()
    elapsed = round((time.perf_counter() - t0) * 1000, 1)
    log.append({"etapa": nome, "ms": elapsed, "ts": datetime.now().isoformat()})
    print(f"  ✓ {nome:<35} {elapsed:>7.1f} ms")
    return resultado

print("═" * 55)
print("PIPELINE BD → PANDAS → FEATURE ENGINEERING → BD")
print("═" * 55)

# ── ETAPA 1: Extrair pedidos ──────────────────────────────────────
def extrair():
    df = pd.read_sql(
        sql="""
            SELECT p.id, p.cliente_id, p.total, p.status,
                   p.criado_em, c.nome AS cliente_nome, c.cidade
            FROM pedidos p
            JOIN clientes c ON c.id = p.cliente_id
        """,
        con=engine,
        parse_dates=["criado_em"],
    )
    return df

df_raw = etapa("Extração do banco", extrair)
print(f"     → {len(df_raw):,} linhas, {df_raw.memory_usage(deep=True).sum()/1e6:.1f} MB")

# ── ETAPA 2: Limpeza ─────────────────────────────────────────────
def limpar():
    df = df_raw.copy()
    df["status"]  = df["status"].str.lower().str.strip()
    df["total"]   = df["total"].fillna(0)
    df["cidade"]  = df["cidade"].fillna("Desconhecida")
    df["ano"]     = df["criado_em"].dt.year
    df["mes"]     = df["criado_em"].dt.month
    return df[df["total"] > 0]

df_clean = etapa("Limpeza e tipagem", limpar)

# ── ETAPA 3: Feature engineering ─────────────────────────────────
def features():
    df = df_clean.groupby("cliente_id").agg(
        cliente_nome   =("cliente_nome", "first"),
        cidade         =("cidade",       "first"),
        total_pedidos  =("id",           "count"),
        ticket_medio   =("total",        "mean"),
        total_gasto    =("total",        "sum"),
        ultimo_pedido  =("criado_em",   "max"),
        qtd_cancelados =("status",       lambda x: (x == "cancelado").sum()),
    ).reset_index()
    df["dias_inativo"]     = (pd.Timestamp("today") - df["ultimo_pedido"]).dt.days
    df["taxa_cancelamento"] = (df["qtd_cancelados"] / df["total_pedidos"] * 100).round(2)
    df["ticket_medio"]     = df["ticket_medio"].round(2)
    df["total_gasto"]      = df["total_gasto"].round(2)
    df["gerado_em"]        = pd.Timestamp("now")
    return df

df_feat = etapa("Feature engineering", features)
print(f"     → {len(df_feat):,} clientes com features")

# ── ETAPA 4: Validação rápida ─────────────────────────────────────
def validar():
    erros = []
    if df_feat["cliente_id"].isnull().any(): erros.append("cliente_id nulo")
    if df_feat["cliente_id"].duplicated().any(): erros.append("cliente_id duplicado")
    if (df_feat["total_gasto"] < 0).any(): erros.append("total_gasto negativo")
    if erros: raise ValueError(f"Validação falhou: {erros}")
    return True

etapa("Validação de qualidade", validar)

# ── ETAPA 5: Gravar no banco ──────────────────────────────────────
def gravar():
    df_feat.to_sql(
        "features_cliente", engine,
        if_exists="replace", index=False,
        chunksize=500, method="multi",
    )
    return True

etapa("Carga no banco (to_sql)", gravar)

# ── Log final ────────────────────────────────────────────────────
with open("pipeline_log.json", "w") as f:
    json.dump(log, f, indent=2)
print("═"*55)
print(f"Pipeline concluído. Log em pipeline_log.json")

Exercício 01

Usando psycopg2, escreva um script que: (a) conecta ao banco via DATABASE_URL do arquivo .env; (b) usa RealDictCursor para buscar os 10 clientes com maior saldo; (c) insere 5 registros de uma só vez na tabela log_eventos usando execute_values; (d) fecha corretamente usando context manager (with). Básico — psycopg2 seguro

import psycopg2

from dotenv import load_dotenv
from datetime import datetime
import os

with psycopg2.connect(os.getenv("DATABASE_URL")) as conn:
    # (a) + (b) RealDictCursor
    with conn.cursor(cursor_factory=RealDictCursor) as cur:
        cur.execute("SELECT id, nome, saldo FROM clientes ORDER BY saldo DESC LIMIT %s", (10,))
        top10 = cur.fetchall()
        print("Top 10 clientes por saldo:")
        for row in top10:
            print(f"  {row['nome']}: R${row['saldo']}")

    # (c) execute_values — 5 registros de uma vez
    with conn.cursor() as cur:
        eventos = [
            (f"evento_{i}", datetime.now())
            for i in range(1, 6)
        ]
        execute_values(
            cur,
            "INSERT INTO log_eventos (evento, ts) VALUES %s",
            eventos,
        )
    conn.commit()
    print("5 eventos inseridos ✓")

Usando SQLAlchemy Core (não ORM), crie um engine com pool_size=3 e escreva uma função buscar_pedidos(status, minimo) que retorna todos os pedidos com o status e total mínimo especificados, usando parâmetros nomeados (:status, :minimo) com text(). A função deve retornar uma lista de dicionários. Teste com pelo menos 2 chamadas diferentes.

Básico — SQLAlchemy Core com parâmetros
Python
from sqlalchemy import create_engine, text
from dotenv import load_dotenv
import os

load_dotenv()
engine = create_engine(os.getenv("DATABASE_URL"), pool_size=3, max_overflow=5)

def buscar_pedidos(status: str, minimo: float) -> list[dict]:
    with engine.connect() as conn:
        resultado = conn.execute(
            text("""
                SELECT id, cliente_id, total, status, criado_em
                FROM pedidos
                WHERE status = :status AND total >= :minimo
                ORDER BY total DESC
            """),
            {"status": status, "minimo": minimo}
        )
        return [dict(row._mapping) for row in resultado]

entregues_altos = buscar_pedidos("entregue", 5000)
print(f"Entregues acima de R$5000: {len(entregues_altos)}")

pendentes = buscar_pedidos("pendente", 0)
print(f"Pendentes: {len(pendentes)}")
for p in pendentes[:3]:
    print(f"  id={p['id']} total=R${p['total']}")
Exercício 03

Use pd.read_sql para carregar a tabela pedidos com parse_dates=["criado_em"]. Em seguida: (a) calcule o ticket médio por mês/ano usando groupby com Grouper(key="criado_em", freq="ME"); (b) filtre apenas pedidos entregues no último ano; (c) grave o resultado em uma tabela chamada metricas_mensais com to_sql usando if_exists="replace", method="multi" e chunksize=500. Imprima as 5 primeiras linhas do resultado.

Médio — pandas read_sql + transformação + to_sql
Python
import pandas as pd
from sqlalchemy import create_engine
from dotenv import load_dotenv
import os

load_dotenv()
engine = create_engine(os.getenv("DATABASE_URL"))

df = pd.read_sql("SELECT * FROM pedidos", engine, parse_dates=["criado_em"])

# (a) Ticket médio por mês
df_mensal = df.groupby(pd.Grouper(key="criado_em", freq="ME")).agg(
    total_pedidos =("id",    "count"),
    ticket_medio  =("total", "mean"),
    receita_total =("total", "sum"),
).reset_index()
df_mensal.rename(columns={"criado_em": "mes_ref"}, inplace=True)
df_mensal["ticket_medio"]  = df_mensal["ticket_medio"].round(2)
df_mensal["receita_total"] = df_mensal["receita_total"].round(2)

# (b) Filtrar último ano
corte = pd.Timestamp("today") - pd.DateOffset(years=1)
df_mensal = df_mensal[df_mensal["mes_ref"] >= corte]

# (c) Gravar no banco
df_mensal.to_sql("metricas_mensais", engine,
                 if_exists="replace", index=False,
                 chunksize=500, method="multi")

print("Métricas mensais gravadas:")
print(df_mensal.head().to_string(index=False))
Exercício 04

Demonstre a diferença de performance entre inserir 10.000 linhas linha a linha (loop com execute) versus em batch com execute_values (psycopg2) e versus to_sql com method="multi" (pandas). Meça o tempo de cada abordagem com time.perf_counter(), imprima uma tabela comparativa e identifique o ganho percentual entre a abordagem mais lenta e a mais rápida.

Médio — Benchmark de inserção em lote
Python
import psycopg2, pandas as pd, time, os
from psycopg2.extras import execute_values
from sqlalchemy import create_engine
from dotenv import load_dotenv

load_dotenv()
DB = os.getenv("DATABASE_URL")
engine = create_engine(DB)
N = 10_000
dados_lista = [(f"evento_{i}",) for i in range(N)]
dados_df = pd.DataFrame({"evento": [f"evento_{i}" for i in range(N)]})

resultados = {}

# 1. Linha a linha
conn = psycopg2.connect(DB); conn.autocommit = False; cur = conn.cursor()
cur.execute("TRUNCATE log_eventos")
t0 = time.perf_counter()
for d in dados_lista:
    cur.execute("INSERT INTO log_eventos (evento) VALUES (%s)", d)
conn.commit(); cur.close(); conn.close()
resultados["Linha a linha"] = (time.perf_counter() - t0) * 1000

# 2. execute_values
conn = psycopg2.connect(DB); conn.autocommit = False; cur = conn.cursor()
cur.execute("TRUNCATE log_eventos")
t0 = time.perf_counter()
execute_values(cur, "INSERT INTO log_eventos (evento) VALUES %s", dados_lista, page_size=500)
conn.commit(); cur.close(); conn.close()
resultados["execute_values"] = (time.perf_counter() - t0) * 1000

# 3. to_sql multi
conn = psycopg2.connect(DB); conn.autocommit = False; cur = conn.cursor()
cur.execute("TRUNCATE log_eventos")
conn.commit(); cur.close(); conn.close()
t0 = time.perf_counter()
dados_df.to_sql("log_eventos", engine, if_exists="append", index=False,
                chunksize=500, method="multi")

Exercício 05 — Desafio

Construa um módulo Python pipeline_completo.py que: (1) lê a tabela pedidos via read_sql usando uma engine com connection pool; (2) valida qualidade com pelo menos 3 checagens (nulos, duplicatas de id, total positivo) — interrompe com log de erro se falhar; (3) calcula features de cliente (ticket médio, dias inativo, taxa de cancelamento, segmento: "VIP" se total_gasto > 10k, "Regular" se > 2k, "Novo" caso contrário); (4) grava em features_cliente_v2 via to_sql; (5) registra num arquivo JSON a hora de início, fim, quantidade de registros em cada etapa e status de cada validação. O módulo deve funcionar invocado diretamente (python pipeline_completo.py) e ser importável sem efeitos colaterais.

Desafio — Pipeline modular e rastreável
Python — pipeline_completo.py
"""pipeline_completo.py — BD → pandas → features → BD com rastreabilidade."""
import pandas as pd, json, os
from sqlalchemy import create_engine
from dotenv import load_dotenv
from datetime import datetime

load_dotenv()

def get_engine():
    url = os.getenv("DATABASE_URL")
    if not url: raise EnvironmentError("DATABASE_URL ausente")
    return create_engine(url, pool_size=5, max_overflow=5, pool_recycle=1800)

def validar(df) -> list[dict]:
    checks = []
    checks.append({"nome": "id_nulos",       "ok": not df["id"].isnull().any()})
    checks.append({"nome": "id_duplicados",   "ok": not df["id"].duplicated().any()})
    checks.append({"nome": "total_positivo",  "ok": (df["total"] > 0).all()})
    falhas = [c["nome"] for c in checks if not c["ok"]]
    if falhas: raise ValueError(f"Validação falhou: {falhas}")
    return checks

def segmento(total):
    if   total > 10_000: return "VIP"
    elif total > 2_000:  return "Regular"
    else:                 return "Novo"

def rodar_pipeline():
    relatorio = {"inicio": datetime.now().isoformat(), "etapas": [], "status": "ok"}
    engine = get_engine()

    try:
        # 1. Extrair
        df = pd.read_sql("SELECT * FROM pedidos", engine, parse_dates=["criado_em"])
        relatorio["etapas"].append({"nome": "extracao", "registros": len(df)})

        # 2. Validar
        checks = validar(df)
        relatorio["etapas"].append({"nome": "validacao", "checks": checks})

        # 3. Features
        feat = df.groupby("cliente_id").agg(
            total_pedidos  =("id",        "count"),
            ticket_medio   =("total",     "mean"),
            total_gasto    =("total",     "sum"),
            ultimo_pedido  =("criado_em", "max"),
            qtd_cancelados =("status",    lambda x: (x=="cancelado").sum()),
        ).reset_index()
        feat["dias_inativo"]     = (pd.Timestamp("today") - feat["ultimo_pedido"]).dt.days
        feat["taxa_cancelamento"] = (feat["qtd_cancelados"]/feat["total_pedidos"]*100).round(2)
        feat["ticket_medio"]     = feat["ticket_medio"].round(2)
        feat["total_gasto"]      = feat["total_gasto"].round(2)
        feat["segmento"]         = feat["total_gasto"].apply(segmento)
        feat["gerado_em"]        = pd.Timestamp("now")
        relatorio["etapas"].append({"nome": "features", "clientes": len(feat),
                                       "segmentos": feat["segmento"].value_counts().to_dict()})

        # 4. Gravar
        feat.to_sql("features_cliente_v2", engine, if_exists="replace",
                    index=False, chunksize=500, method="multi")
        relatorio["etapas"].append({"nome": "carga", "tabela": "features_cliente_v2"})

    except Exception as e:
        relatorio["status"] = "erro"
        relatorio["erro"] = str(e)
        print(f"✗ Pipeline falhou: {e}")

    relatorio["fim"] = datetime.now().isoformat()
    with open("relatorio_pipeline.json", "w") as f:
        json.dump(relatorio, f, indent=2, ensure_ascii=False)
    print(f"Pipeline: {relatorio['status'].upper()} — relatório em relatorio_pipeline.json")
    return relatorio

if __name__ == "__main__":
    rodar_pipeline()
