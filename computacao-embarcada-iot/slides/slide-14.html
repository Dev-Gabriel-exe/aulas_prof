Wake Word Detection — como o ESP32 "ouve" uma palavra

Wake word (ou keyword spotting) é o caso de uso mais icônico de TinyML: o dispositivo fica em modo de baixo consumo ouvindo continuamente, e acorda o sistema maior apenas quando detecta uma palavra específica. "Hey Siri", "Ok Google", "Alexa" — todos rodam modelos TinyML no próprio chip de áudio.

O segredo é não processar áudio bruto. Em vez disso, o pipeline extrai um espectrograma Mel (MFCC) — uma representação 2D do áudio que captura as frequências mais relevantes para a percepção humana — e o modelo classifica essa "imagem" de áudio.

🙋 Por que o pipeline de wake word extrai MFCC em vez de usar as amostras de áudio bruto diretamente como entrada do modelo?
A
Porque TFLite Micro não suporta entrada de áudio bruto — só aceita features pré-processadas
B
Porque 1 segundo de áudio a 16kHz gera 16.000 amostras — um modelo que aceita 40 coeficientes MFCC é 400× menor, cabe na RAM do MCU e é muito mais rápido de inferir
C
Porque áudio bruto é analógico e precisa ser convertido para formato digital antes de qualquer ML
D
Porque MFCC remove o ruído de fundo automaticamente, tornando o modelo mais robusto

Classificação de gestos com acelerômetro MPU6050

O MPU6050 é um IMU (Inertial Measurement Unit) de 6 eixos: acelerômetro 3D + giroscópio 3D, comunicação I2C, endereço 0x68 (ou 0x69). É o sensor de movimento mais usado em projetos TinyML de gestos — presente em controles de game, sistemas de detecção de queda e interfaces sem toque.

Para classificar gestos, você coleta uma janela curta de dados (ex: 2 segundos a 100Hz = 200 amostras × 6 eixos = 1.200 valores) e extrai features que resumem o movimento — a rede aprende a distinguir "circular" de "sacudir" de "inclinar" por meio dessas features.

Arduino C — coletar janela MPU6050 e extrair features
#include <Wire.h>
#include <MPU6050_light.h>   // instalar: "MPU6050_light" by rfetick
#include <math.h>

#define JANELA_N   50    // 50 amostras @ ~50Hz = ~1 segundo
#define N_FEATURES 6    // média e std de acX, acY, acZ

MPU6050 mpu(Wire);
float  buf_ax[JANELA_N], buf_ay[JANELA_N], buf_az[JANELA_N];
float  features[N_FEATURES];

float calcMedia(float* buf) {
  float s = 0;
  for(int i=0;i<JANELA_N;i++) s += buf[i];
  return s / JANELA_N;
}

float calcStd(float* buf, float media) {
  float var = 0;
  for(int i=0;i<JANELA_N;i++) var += (buf[i]-media)*(buf[i]-media);
  return sqrt(var/JANELA_N);
}

void coletarJanela() {
  for(int i=0;i<JANELA_N;i++) {
    mpu.update();
    buf_ax[i] = mpu.getAccX();
    buf_ay[i] = mpu.getAccY();
    buf_az[i] = mpu.getAccZ();
    delay(20);   // ~50Hz
  }
  // Extrair features: [media_x, std_x, media_y, std_y, media_z, std_z]
  float mx=calcMedia(buf_ax), my=calcMedia(buf_ay), mz=calcMedia(buf_az);
  features[0]=mx; features[1]=calcStd(buf_ax,mx);
  features[2]=my; features[3]=calcStd(buf_ay,my);
  features[4]=mz; features[5]=calcStd(buf_az,mz);
}

void setup() {
  Serial.begin(115200);
  Wire.begin();
  mpu.begin();
  mpu.calcOffsets();   // calibrar — mantenha parado por 3s
  Serial.println("MPU6050 pronto!");
}

void loop() {
  coletarJanela();
  // Aqui: alimentar features[] no modelo TFLite e inferir
  Serial.printf("mx=%.2f sx=%.2f my=%.2f sy=%.2f mz=%.2f sz=%.2f\n",
    features[0],features[1],features[2],
    features[3],features[4],features[5]);
}
Detecção de anomalias em série temporal — a abordagem do LAB

Detecção de anomalia em dados de sensores é o caso de uso TinyML mais comum em IoT industrial. O princípio: o modelo aprende o padrão "normal" durante o treino e, durante a inferência, classifica cada janela como Normal ou Anomalia com base em features da janela.

A abordagem desta aula usa um modelo de classificação binária treinado com features estatísticas de janelas de temperatura. É mais simples que um autoencoder (que também é válido) e cabe em poucos KB de flash.

Série temporal de temperatura — janela de 10 leituras com anomalia
Normal Anomalia detectada Threshold (μ + 2σ)
Python — treinar classificador de anomalia (DHT22)
import numpy as np
import tensorflow as tf

# Features por janela: [media, std, min, max, amplitude, rms]
def extrair_features(janela):
    m = np.mean(janela)
    s = np.std(janela)
    return [m, s, np.min(janela), np.max(janela),
            np.max(janela) - np.min(janela),
            np.sqrt(np.mean(np.square(janela)))]

# Dados simulados — substituir por leituras reais do DHT22
# Classe 0 = Normal (T entre 20–28°C, variação pequena)
# Classe 1 = Anomalia (T fora do range ou variação brusca)
normais   = [np.random.normal(24, 1.5, 10) for _ in range(60)]
anomalias = [np.random.normal(24, 1.5, 10) + np.random.choice([8,-8])
             for _ in range(40)]

X = np.array([extrair_features(j) for j in normais + anomalias], dtype=np.float32)
y = np.array([0]*60 + [1]*40)

# Normalizar — guardar média e desvio para usar no ESP32
X_mean, X_std = X.mean(axis=0), X.std(axis=0) + 1e-8
X_norm = (X - X_mean) / X_std
print(f"Média: {X_mean.round(3)}")
print(f"Std:   {X_std.round(3)}")

# Modelo pequeno — 6 features → 8 → 4 → 2 classes
model = tf.keras.Sequential([
    tf.keras.layers.Dense(8, activation='relu', input_shape=(6,)),
    tf.keras.layers.Dense(4, activation='relu'),
    tf.keras.layers.Dense(2, activation='softmax')
])
model.compile(optimizer='adam',
              loss='sparse_categorical_crossentropy',
              metrics=['accuracy'])
model.fit(X_norm, y, epochs=300, verbose=0)
loss, acc = model.evaluate(X_norm, y, verbose=0)
print(f"Acurácia: {acc:.2%}")

# Converter e quantizar para int8
converter = tf.lite.TFLiteConverter.from_keras_model(model)
converter.optimizations = [tf.lite.Optimize.DEFAULT]
def rep_data():
    for x in X_norm[:20]: yield [x.reshape(1,6).astype(np.float32)]
converter.representative_dataset = rep_data
converter.target_spec.supported_ops = [tf.lite.OpsSet.TFLITE_BUILTINS_INT8]
converter.inference_input_type  = tf.int8
converter.inference_output_type = tf.int8
tflite = converter.convert()
with open("modelo_anomalia.tflite","wb") as f: f.write(tflite)
print(f"Modelo salvo: {len(tflite)} bytes")
🙋 Por que usamos features estatísticas (média, desvio, RMS) em vez de passar a janela de 10 leituras brutas diretamente ao modelo TinyML?
A
Porque o TFLite Micro não suporta vetores de entrada com mais de 6 elementos
B
Features resumem o padrão da janela em poucos valores informativos — o modelo fica menor, mais rápido e mais generalizável; 10 leituras brutas aumentam a dimensão sem acrescentar informação nova
C
Porque o sensor DHT22 já entrega diretamente features estatísticas em vez de leituras brutas
D
Para economizar memória flash — 6 floats ocupam menos que 10 floats na flash

Arduino C — detector de anomalia completo: TFLite + DHT22 + MQTT
#include <DHT.h>
#include <WiFi.h>
#include <PubSubClient.h>
#include <ArduinoJson.h>
#include "TensorFlowLite_ESP32.h"
#include "tensorflow/lite/micro/all_ops_resolver.h"
#include "tensorflow/lite/micro/micro_interpreter.h"
#include "tensorflow/lite/schema/schema_generated.h"
#include "modelo_anomalia.h"
#include <math.h>

// ── Configurações ──
const char* SSID      = "NomeDaRede";
const char* SENHA     = "SenhaDaRede";
const char* BROKER_IP = "192.168.1.100";
#define PINO_DHT   4
#define JANELA_N   10
#define ARENA_SIZE 6144

// ── Parâmetros de normalização — copiar do treino Python ──
const float MEAN[6] = {24.1f, 1.4f, 22.0f, 26.5f, 4.5f, 24.2f};
const float STD[6]  = {2.3f,  1.1f, 2.4f,  2.6f,  1.8f, 2.3f};

DHT dht(PINO_DHT, DHT22);
WiFiClient wc; PubSubClient mqtt(wc);

// TFLM
const tflite::Model*       model;
tflite::MicroInterpreter*  interpreter;
TfLiteTensor*              input;
TfLiteTensor*              output;
uint8_t tensor_arena[ARENA_SIZE];

// Janela deslizante
float janela[JANELA_N]; int idx=0; bool cheia=false;

void extrairFeatures(float* feat) {
  float m=0, mn=janela[0], mx=janela[0], sq=0;
  for(int i=0;i<JANELA_N;i++){
    m+=janela[i]; sq+=janela[i]*janela[i];
    if(janela[i]<mn) mn=janela[i];
    if(janela[i]>mx) mx=janela[i];
  }
  m/=JANELA_N; sq/=JANELA_N;
  float var=0;
  for(int i=0;i<JANELA_N;i++) var+=(janela[i]-m)*(janela[i]-m);
  feat[0]=m; feat[1]=sqrt(var/JANELA_N); feat[2]=mn;
  feat[3]=mx; feat[4]=mx-mn; feat[5]=sqrt(sq);
}

int8_t quantizar(float v, float sc, int zp) {
  return (int8_t)constrain((int)roundf(v/sc)+zp, -128, 127);
}

void publicarAlerta(float t, float conf) {
  if(WiFi.status()!=WL_CONNECTED||!mqtt.connected()) return;
  StaticJsonDocument<96> doc;
  doc["temp"]=t; doc["confianca"]=conf; doc["alerta"]="ANOMALIA";
  char buf[96]; serializeJson(doc,buf);
  mqtt.publish("borda/anomalia/temperatura",buf);
}

void setup() {
  Serial.begin(115200); dht.begin();

  // Wi-Fi
  WiFi.begin(SSID,SENHA);
  while(WiFi.status()!=WL_CONNECTED) delay(300);
  mqtt.setServer(BROKER_IP,1883);
  mqtt.connect("esp32-anomalia");

  // TFLite
  model = tflite::GetModel(modelo_anomalia);
  static tflite::AllOpsResolver resolver;
  static tflite::MicroInterpreter si(model,resolver,tensor_arena,ARENA_SIZE);
  interpreter=&si;
  if(interpreter->AllocateTensors()!=kTfLiteOk){
    Serial.println("AllocateTensors falhou! Aumente ARENA_SIZE."); while(true);
  }
  input=interpreter->input(0); output=interpreter->output(0);
  Serial.println("Pronto. Acumulando janela...");
}

void loop() {
  mqtt.loop();

  float t=dht.readTemperature();
  if(isnan(t)){delay(2000);return;}

  // Acumular janela deslizante
  janela[idx]=(t-0); // raw temp
  idx=(idx+1)%JANELA_N;
  if(idx==0) cheia=true;
  if(!cheia){delay(3000);return;}

  // Extrair e normalizar features
  float feat[6]; extrairFeatures(feat);
  float sc=input->params.scale; int zp=input->params.zero_point;
  for(int i=0;i<6;i++)
    input->data.int8[i] = quantizar((feat[i]-MEAN[i])/STD[i], sc, zp);

  // Inferência
  interpreter->Invoke();

  float osc=output->params.scale; int ozp=output->params.zero_point;
  float pNormal   = (output->data.int8[0]-ozp)*osc;
  float pAnomalia = (output->data.int8[1]-ozp)*osc;

  Serial.printf("T=%.1f°C  Normal=%.2f  Anomalia=%.2f  → %s\n",
    t, pNormal, pAnomalia, pAnomalia>0.7f?"*** ANOMALIA ***":"Normal");

  if(pAnomalia > 0.7f) publicarAlerta(t, pAnomalia);

  delay(3000);
}

Exercício 01 — Calibrar o threshold de alerta

O código usa threshold fixo de 0.7 para classificar uma janela como anomalia. Modifique para usar um threshold configurável via MQTT: ao receber uma mensagem no tópico borda/config/threshold com um valor float (ex: "0.85"), o ESP32 atualiza o threshold em tempo de execução e confirma publicando o novo valor em borda/config/threshold/ack. Isso simula o gerenciamento remoto de um modelo em campo.

LAB · Integração MQTT
Trecho — threshold via MQTT
float thresholdAnomalia = 0.70f;   // variável global

void callbackMQTT(char* topico, byte* payload, unsigned int tam) {
  String msg = "";
  for(int i=0;i<(int)tam;i++) msg+=(char)payload[i];

  if(String(topico)=="borda/config/threshold") {
    float novo = msg.toFloat();
    if(novo>0.0f && novo<1.0f) {
      thresholdAnomalia = novo;
      char buf[32];
      snprintf(buf, sizeof(buf), "%.2f", thresholdAnomalia);
      mqtt.publish("borda/config/threshold/ack", buf);
      Serial.printf("Threshold atualizado: %.2f\n", thresholdAnomalia);
    }
  }
}

// No setup(), após mqtt.connect():
mqtt.setCallback(callbackMQTT);
mqtt.subscribe("borda/config/threshold");
Exercício 02 — Adicionar umidade como segunda feature

O modelo atual usa apenas temperatura. Expanda o pipeline para incluir umidade como segunda variável: a janela deslizante agora acumula temperatura E umidade em paralelo. O vetor de features sobe para 12 (6 por variável). Treine um novo modelo com os dados expandidos, gere um novo .h, e integre ao ESP32. Compare a acurácia do modelo com 6 vs 12 features. Quando a umidade melhora a detecção e quando não faz diferença?

LAB + Análise
Análise esperada
  • A umidade melhora a detecção quando a anomalia envolve mudança simultânea de temperatura e umidade (ex: incêndio → temperatura sobe, umidade cai drasticamente).
  • A umidade não ajuda quando a anomalia é puramente térmica (ex: superaquecimento de um componente eletrônico), pois a umidade não varia.
  • O custo: tensor arena maior, modelo ligeiramente maior. Para o ESP32, ainda viável — ambos cabem confortavelmente na flash e RAM disponíveis.

// Pressione o botão para iniciar a coleta de 1 janela
// Imprime: classe,mx,sx,my,sy,mz,sz
#define BTN_PIN   0    // BOOT button no ESP32
int classeAtual = 0;   // mudar manualmente: 0=sacudir, 1=inclinar, 2=circular

void loop() {
  if (digitalRead(BTN_PIN) == LOW) {
    delay(50);   // debounce
    Serial.print("Prepare o gesto "); Serial.print(classeAtual);
    Serial.println("... 3s");
    delay(3000);
    coletarJanela();
    Serial.printf("%d,%.3f,%.3f,%.3f,%.3f,%.3f,%.3f\n",
      classeAtual,

      features[3],features[4],features[5]);
  }
}
Análise típica de matriz de confusão
  • "Sacudir" tem std muito alta em todos os eixos — fácil de distinguir.
  • "Inclinar para frente" tem média_y deslocada — bem caracterizado.
  • "Circular" é o mais difícil: features médias ficam próximas do repouso pois a energia se distribui em todos os eixos. Adicionar FFT ou energia por eixo melhora significativamente.

from datetime import datetime
from collections import deque

BROKER = "192.168.1.100"
historico = deque(maxlen=3600)   # máximo 1h de anomalias
ultima_anomalia = None

def on_message(client, ud, msg):
    global ultima_anomalia
    try:
        d = json.loads(msg.payload.decode())
        ts = datetime.now()
        historico.append({"ts": ts, "temp": d.get("temp",0)})
        ultima_anomalia = ts
        print(f"[{ts.strftime('%H:%M:%S')}] Anomalia: T={d.get('temp')}°C conf={d.get('confianca'):.2f}")
    except: pass

def publicar_resumo(client):
    while True:
        time.sleep(300)
        agora = datetime.now()
        recentes = [e for e in historico
                    if (agora - e["ts"]).seconds < 3600]
        temp_max = max((e["temp"] for e in recentes), default=0)
        resumo = {
            "anomalias_ultima_hora": len(recentes),
            "temp_max_anomalia": temp_max,
            "ultima_anomalia_ts": ultima_anomalia.strftime("%H:%M:%S") if ultima_anomalia else "nenhuma"
        }
        client.publish("dashboard/resumo", json.dumps(resumo))
        print(f"Resumo publicado: {resumo}")

c = mqtt.Client()
c.on_connect = lambda cl,u,f,rc: cl.subscribe("borda/anomalia/#")
c.on_message = on_message
c.connect(BROKER, 1883)
threading.Thread(target=publicar_resumo, args=(c,), daemon=True).start()
c.loop_forever()
